# Upper case 
x = "my name is awais and my father name is nazim"
print(x.upper())

#Lowercase
x = "MY NAME IS awais AND MY FATHER NAME IS nazim"
print(x.lower())

# Removing Whitespace

x = "   Awais "
print(x.strip())  

# Replace string 

x = "This is data  science tutorials"
print(x.replace("This" , "Bahi"))

# Split string 

x = "This, is, data , science, tutorials"
print(x.split(","))
